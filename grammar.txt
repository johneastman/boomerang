statement:
    let IDENTIFIER = expression
    func IDENTIFIER (expression, expression, ...) { statement; statement; ... };
    if (expression) { statement; statement; ... } else { statement; statement; ... };
    expression

expression:
    addition [==|!=|>=|<=|>|<] addition

addition:
    term +|- term

term:
    factor *|/ factor

factor:
    !|+|- expression
    number
    boolean
    ( expression )
    statement
    return expression

number:
    0 - 9

boolean:
    true|false
